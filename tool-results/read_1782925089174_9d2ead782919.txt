   340→export default function GuardrailsGateway() {
   341→  const [prompt, setPrompt] = useState("");
   342→  const [isProcessing, setIsProcessing] = useState(false);
   343→  const [result, setResult] = useState<GatewayResult | null>(null);
   344→  const [dashboard, setDashboard] = useState<DashboardData | null>(null);
   345→  const [activeResultType, setActiveResultType] = useState<
   346→    "dry" | "full" | null
   347→  >(null);
   348→  const [piiOpen, setPiiOpen] = useState(true);
   349→  const [injectionOpen, setInjectionOpen] = useState(true);
   350→
   351→  // ── Fetch dashboard data ──
   352→  const fetchDashboard = useCallback(async () => {
   353→    try {
   354→      const res = await fetch("/api/dashboard");
   355→      if (res.ok) {
   356→        const data = await res.json();
   357→        setDashboard(data);
   358→      }
   359→    } catch {
   360→      // silently fail
   361→    }
   362→  }, []);
   363→
   364→  useEffect(() => {
   365→    fetchDashboard();
   366→    const interval = setInterval(fetchDashboard, 5000);
   367→    return () => clearInterval(interval);
   368→  }, [fetchDashboard]);
   369→
   370→  // ── Submit handler ──
   371→  const handleSubmit = useCallback(
   372→    async (dryRun: boolean) => {
   373→      if (!prompt.trim()) {
   374→        toast.error("Please enter a prompt to analyze.");
   375→        return;
   376→      }
   377→
   378→      setIsProcessing(true);
   379→      setResult(null);
   380→      setActiveResultType(null);
   381→
   382→      try {
   383→        // Step 1: Always run input checks via server (dry run)
   384→        const dryRes = await fetch("/api/gateway", {
   385→          method: "POST",
   386→          headers: { "Content-Type": "application/json" },
   387→          body: JSON.stringify({ input: prompt, dryRun: true }),
   388→        });
   389→
   390→        if (!dryRes.ok) throw new Error("Gateway request failed");
   391→        const dryData: GatewayResult = await dryRes.json();
   392→
   393→        // If blocked or dry-run only, return early
   394→        if (dryRun || dryData.inputStatus?.status === "blocked") {
   395→          setResult(dryData);
   396→          setActiveResultType("dry");
   397→          setTimeout(fetchDashboard, 500);
   398→          return;
   399→        }
   400→
   401→        // Step 2: For full pipeline, server handles the LLM simulation
   402→        const fullRes = await fetch("/api/gateway", {
   403→          method: "POST",
   404→          headers: { "Content-Type": "application/json" },
   405→          body: JSON.stringify({ input: prompt }),
   406→        });
   407→
   408→        if (!fullRes.ok) throw new Error("Gateway request failed");
   409→        const fullData: GatewayResult = await fullRes.json();
   410→        setResult(fullData);
   411→        setActiveResultType("full");
   412→        setTimeout(fetchDashboard, 500);
   413→      } catch (err) {
   414→        toast.error(
   415→          err instanceof Error ? err.message : "Request failed"
   416→        );
   417→      } finally {
   418→        setIsProcessing(false);
   419→      }
   420→    },
   421→    [prompt, fetchDashboard]
   422→  );
   423→
   424→  // ── Example prompt click ──
   425→  const handleExampleClick = useCallback(
   426→    (examplePrompt: string) => {
   427→      setPrompt(examplePrompt);
   428→      // Auto-submit as dry run after setting the prompt
   429→      setTimeout(() => {
   430→        handleSubmit(true);
   431→      }, 100);
   432→    },
   433→    [handleSubmit]
   434→  );
   435→
   436→  // ── Derive display status ──
   437→  const displayStatus = result
   438→    ? result.status === "blocked_input" || result.status === "blocked_output"
   439→      ? "blocked"
   440→      : result.status === "modified"
   441→        ? "modified"
   442→        : "allowed"
   443→    : null;
   444→
   445→  // ── Chart data ──
   446→  const chartData = (dashboard?.blockedBy || []).map((b, i) => ({
   447→    type: b.type.replace(/_/g, " ").replace(/\b\w/g, (c) => c.toUpperCase()),
   448→    count: b.count,
   449→    fill: CHART_COLORS[i % CHART_COLORS.length],
   450→  }));
   451→
   452→  return (
   453→    <div className="min-h-screen flex flex-col" style={{ backgroundColor: "#09090b" }}>
   454→      {/* ─── HEADER ───────────────────────────────────────────────── */}
   455→      <header
   456→        className="sticky top-0 z-50 border-b border-white/[0.06] backdrop-blur-xl"
   457→        style={{ backgroundColor: "rgba(9,9,11,0.85)" }}
   458→      >
   459→        <div className="mx-auto flex h-16 max-w-7xl items-center justify-between px-4 sm:px-6">
   460→          {/* Left */}
   461→          <div className="flex items-center gap-3">
   462→            <div className="flex h-9 w-9 items-center justify-center rounded-lg bg-red-500/10 border border-red-500/20">
   463→              <Shield className="h-5 w-5 text-red-400" />
   464→            </div>
   465→            <div>
   466→              <h1 className="text-sm font-bold tracking-tight text-zinc-100 sm:text-base">
   467→                LLM Guardrails Gateway
   468→              </h1>
   469→              <p className="hidden text-xs text-zinc-500 sm:block">
   470→                Input/Output Safety Layer for LLM Applications
   471→              </p>
   472→            </div>
   473→          </div>
   474→
   475→          {/* Right — Live Stats */}
   476→          <div className="flex items-center gap-2 sm:gap-3">
   477→            <div className="hidden items-center gap-1.5 rounded-md border border-white/[0.06] bg-white/[0.03] px-2.5 py-1.5 sm:flex">
   478→              <Activity className="h-3.5 w-3.5 text-zinc-500" />
   479→              <span className="text-xs text-zinc-400">Total</span>
   480→              <span className="text-xs font-bold tabular-nums text-zinc-200">
   481→                {dashboard?.stats.total ?? "—"}
   482→              </span>
   483→            </div>
   484→            <div className="hidden items-center gap-1.5 rounded-md border border-white/[0.06] bg-white/[0.03] px-2.5 py-1.5 sm:flex">
   485→              <ShieldX className="h-3.5 w-3.5 text-red-500/70" />
   486→              <span className="text-xs text-zinc-400">Block</span>
   487→              <span
   488→                className={`text-xs font-bold tabular-nums ${
   489→                  dashboard && Number(dashboard.stats.blockRate) > 20
   490→                    ? "text-red-400"
   491→                    : "text-zinc-200"
   492→                }`}
   493→              >
   494→                {dashboard?.stats.blockRate ?? "—"}%
   495→              </span>
   496→            </div>

   517→      <div
   518→        className="border-b border-white/[0.06]"
   519→        style={{ backgroundColor: "#0c0c0f" }}

   524→              Rules
   525→            </span>
   526→            {POLICY_RULES.map((rule) => (
   527→              <motion.div
   528→                key={rule.type}
   529→                whileHover={{ scale: 1.04 }}
   530→                whileTap={{ scale: 0.97 }}

   532→                style={{
   533→                  borderColor: rule.enabled
   534→                    ? "rgba(16,185,129,0.2)"
   535→                    : "rgba(239,68,68,0.2)",
   536→                  backgroundColor: rule.enabled

   538→                    : "rgba(239,68,68,0.06)",

   548→                />
   549→                {rule.name}
   550→                {rule.enabled && (

   559→        </div>

   562→      {/* ─── MAIN AREA ───────────────────────────────────────────── */}
   563→      <main className="mx-auto w-full max-w-7xl flex-1 px-4 py-6 sm:px-6">
   564→        <div className="grid gap-6 lg:grid-cols-12">
   565→          {/* ── LEFT COLUMN ── */}
   566→          <div className="flex flex-col gap-6 lg:col-span-8">
   567→            {/* Section A: Input Panel */}
   568→            <Card className="border-white/[0.06] bg-[#0c0c0f] py-4">

   584→                  className="min-h-[100px] resize-y border-white/[0.08] bg-[#07070a] font-mono text-sm text-zinc-300 placeholder:text-zinc-700 focus-visible:ring-red-500/30 focus-visible:border-red-500/40"
