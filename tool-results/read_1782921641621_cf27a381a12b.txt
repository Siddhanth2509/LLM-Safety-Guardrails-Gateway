   937→// ─── Pipeline Visualization Component ────────────────────────────────
   938→
   939→function PipelineVisualization({
   940→  activeStep,
   941→}: {
   942→  activeStep: string | null;
   943→}) {
   944→  const steps = ["retrieving", "generating", "critiquing", "accepted"];
   945→
   946→  const getStepState = (step: string) => {
   947→    if (step === activeStep) return "active";
   948→    const currentIdx = steps.indexOf(activeStep || "");
   949→    const stepIdx = steps.indexOf(step);
   950→    if (stepIdx < currentIdx) return "completed";
   951→    return "pending";
   952→  };
   953→
   954→  return (
   955→    <div className="flex items-center justify-between gap-2 px-2">
   956→      {steps.map((step, idx) => {
   957→        const state = getStepState(step);
   958→        const config = STEP_CONFIG[step];
   959→        return (
   960→          <div key={step} className="flex items-center gap-2 flex-1">
   961→            <div
   962→              className={`flex items-center gap-1.5 px-3 py-1.5 rounded-full text-xs font-medium transition-all duration-300 ${
   963→                state === "active"
   964→                  ? "bg-primary/10 border border-primary/20 text-primary"
   965→                  : state === "completed"
   966→                    ? "bg-emerald-500/10 border border-emerald-500/20 text-emerald-400"
   967→                    : "bg-muted/30 border border-border/20 text-muted-foreground/40"
   968→              }`}
   969→            >
   970→              {state === "active" ? (
   971→                <Loader2 className="w-3 h-3 animate-spin" />
   972→              ) : state === "completed" ? (
   973→                <CheckCircle2 className="w-3 h-3" />
   974→              ) : (
   975→                <span className="w-3 h-3">{config?.icon}</span>
   976→              )}
   977→              {config?.label || step}
   978→            </div>
   979→            {idx < steps.length - 1 && (
   980→              <ArrowRight
   981→                className={`w-3 h-3 shrink-0 transition-colors ${
   982→                  state === "completed"
   983→                    ? "text-emerald-400/50"
   984→                    : "text-muted-foreground/20"
   985→                }`}
   986→              />
   987→            )}
   988→          </div>
   989→        );
   990→      })}
   991→    </div>
   992→  );
